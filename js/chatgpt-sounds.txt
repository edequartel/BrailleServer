  <script>
    // Optional external logger
    function logMessage(msg) {
      console.log("[PAGE]", msg);
    }

    window.addEventListener("DOMContentLoaded", async () => {
      // We are in /pages → config is at ../config/sounds.json
      await Sounds.init("../config/sounds.json", logMessage);

      // Example: play A, bal, click, correct
      document.getElementById("btnA").onclick = () => Sounds.playLetter("nl", "a");
      document.getElementById("btnBal").onclick = () => Sounds.playWord("nl", "bal");
      document.getElementById("btnClick").onclick = () => Sounds.playUI("nl", "click");
      document.getElementById("btnCorrect").onclick = () => Sounds.playShared("correct");
    });
